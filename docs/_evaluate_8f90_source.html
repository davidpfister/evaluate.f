159 subroutine eval_c8(expr, val, ierr)
160 character(*), intent(in) :: expr
161 complex(c8), intent(out) :: val
162 integer, intent(out), optional :: ierr
163 !private
164 character(len(expr) + 1) :: tempstr
165 character :: cop
166 integer :: isp(numtok) ! On stack priority of operators in opstack
167 integer :: lstr
168 complex(c8) :: cval, oper1, oper2
169 real(r8) :: valr, vali
170 type(item):: token(numtok) ! List of tokens ( a token is an operator or
171 ! operand) in postfix order
172 type(item) :: x, junk, tok
173 integer :: i, icp, insp, isum, ntok, ios
174
175 if (present(ierr)) ierr = 0
176 token(1:)%char = ' '
177
178 if (len_trim(expr) == 0) then ! Expression empty
179 if (present(ierr)) ierr = 1
180 write (*, *) 'Error: expression being evaluated is empty'
181 return
182 end if
183
184 tempstr = adjustl(expr)
185 call removesp(tempstr) ! Removes spaces, tabs, and control characters
186
187 ! STEP 1: Convert string to token array. Each token is either an operator or
188 ! an operand. Token array will be in postfix (reverse Polish) order.
189
190 ntok = 0
191 ibin = 0
192 itop = 0
193 do
194 lstr = len_trim(tempstr)
195 call get_next_token(tempstr(1:lstr), tok, icp, insp)
196 select case (tok%type)
197 case ('S')
198 ntok = ntok + 1
199 token(ntok) = tok
200 case ('E')
201 do
202 if (itop < 1) exit
203 call popop(x) ! Output remaining operators on stack
204 ntok = ntok + 1
205 token(ntok) = x
206 end do
207 ntok = ntok + 1
208 token(ntok) = tok
209 exit
210 case ('R') ! Token is right parenenthesis
211 do
212 if (itop <= 0) exit
213 if (opstack(itop)%type == 'L') exit ! Output operators on stack down
214 call popop(x) ! to left parenthesis
215 ntok = ntok + 1
216 token(ntok) = x
217 end do
218 call popop(junk) ! Remove left parenthesis from stack
219 if (itop > 0) then
220 if (opstack(itop)%type == 'F') then ! Output function name if present
221 call popop(x)
222 ntok = ntok + 1
223 token(ntok) = x
224 end if
225 end if
226 case ('D') ! Token is comma
227 do
228 if (itop <= 0) exit
229 if (opstack(itop)%type == 'L') exit ! Output operators on stack down
230 call popop(x) ! to left parenthesis
231 ntok = ntok + 1
232 token(ntok) = x
233 end do
234 case ('U', 'B', 'L', 'F') ! Token is operator, left parenthesis or function name
235 do
236 if (itop <= 0) exit ! Output operators on stack having
237 if (isp(itop) < icp) exit
238 call popop(x) ! an instack priority that is
239 ntok = ntok + 1 ! greater than or equal to the
240 token(ntok) = x ! priority of the incoming operator
241 end do
242 call pushop(tok) ! Put incoming operator on stack
243 isp(itop) = insp
244 end select
245 end do
246
247 isum = 0 ! Error check for matching parentheses
248 do i = 1, ntok
249 if (token(i)%type == 'L') isum = isum + 1
250 if (token(i)%type == 'R') isum = isum - 1
251 end do
252 if (isum /= 0) then
253 if (present(ierr)) ierr = 2
254 write (*, *) 'Error in the evaluation of the expression ', trim(expr)
255 write (*, *) "Parentheses don't match"
256 write (*, *)
257 return
258 end if
259
260 ! STEP 2: Evaluate token string in postfix order
261
262 itop = 0
263 i = 0
264 do while (i < ntok)
265 i = i + 1
266 x = token(i)
267 select case (x%type)
268 case ('E') ! Token is end token
269 if (itop > 1) then
270 if (present(ierr)) ierr = 12
271 write (*, *) 'Error: missing operator in expression ', trim(expr)
272 write (*, *)
273 return
274 end if
275 call popval(val) ! Final result left on stack of values
276 exit
277 case ('S') ! Token is operand
278 call getvalue(x%char, cval, ios) ! Evaluate operand
279 if (ios /= 0) then
280 if (present(ierr)) ierr = ios
281 return
282 end if
283 call pushval(cval) ! Put value of operand on stack
284 case ('B') ! Token is a binary operator
285 if (itop < 2) then
286 if (present(ierr)) ierr = 5
287 write (*, *) 'Error in evaluation of expression ', trim(expr)
288 write (*, *) 'Less than two operands for binary operator ' &
289 , trim(x%char)
290 write (*, *)
291 return
292 end if
293 call popval(oper1) ! Pull off top two values from stack
294 call popval(oper2)
295 select case (trim(x%char)) ! Perform operation on values
296 case ('^')
297 cval = oper2**oper1
298 case ('*')
299 cval = oper2*oper1
300 case ('/')
301 if (oper1 == (0._r8, 0._r8)) then
302 if (present(ierr)) ierr = 10
303 write (*, *) 'Error in expression ', trim(expr)
304 write (*, *) 'Division by zero'
305 write (*, *)
306 return
307 end if
308 cval = oper2/oper1
309 case ('+')
310 cval = oper2 + oper1
311 case ('-')
312 cval = oper2 - oper1
313 end select
314 call pushval(cval) ! Put result back on stack
315 case ('U') ! Token is unary operator
316 if (itop == 0) then
317 if (present(ierr)) ierr = 6
318 write (*, *) 'Error in expression ', trim(expr)
319 write (*, *) 'No operand for unary operator ', trim(x%char)
320 write (*, *)
321 return
322 else
323 call popval(oper1) ! Pull top value off stack
324 end if
325 select case (trim(x%char)) ! Operate on value
326 case ('+')
327 cval = oper1
328 case ('-')
329 cval = -oper1
330 end select
331 call pushval(cval) ! Put result back on stack
332 case ('F') ! Token is a function name
333 if (itop == 0) then
334 if (present(ierr)) ierr = 7
335 write (*, *) 'Error in expression ', trim(expr)
336 write (*, *) 'Missing argument(s) for function ', trim(x%char)
337 write (*, *)
338 return
339 else
340 call popval(oper1) ! Pull top value off stack
341 end if
342 tempstr = uppercase(x%char)
343 select case (trim(tempstr)) ! Evaluate function
344 case ('SIN')
345 cval = sin(oper1)
346 case ('COS')
347 cval = cos(oper1)
348 case ('TAN')
349 oper2 = cos(oper1)
350 if (abs(oper2) == 0.0_r8) then
351 if (present(ierr)) ierr = 14
352 write (*, *) 'Error: argument of tan function a multiple', &
353 ' of pi/2 in expression ', trim(expr)
354 write (*, *)
355 return
356 else
357 cval = sin(oper1)/oper2
358 end if
359 case ('SINH')
360 cval = sinh(oper1)
361 case ('COSH')
362 cval = cosh(oper1)
363 case ('TANH')
364 cval = tanh(oper1)
365 case ('ACOS')
366 cval = acos(oper1)
367 case ('ASIN')
368 cval = asin(oper1)
369 case ('ATAN')
370 cval = atan(oper1)
371 case ('SQRT')
372 if (real(oper1, r8) < 0. .and. aimag(oper1) == 0.) then
373 if (present(ierr)) ierr = 9
374 write (*, *) 'Warning: square root of negative real number', &
375 ' in expression ', trim(expr)
376 write (*, *)
377 end if
378 cval = sqrt(oper1)
379 case ('ABS')
380 cval = abs(oper1)
381 case ('LOG')
382 if (real(oper1, r8) <= 0. .and. aimag(oper1) == 0.) then
383 if (present(ierr)) ierr = 8
384 write (*, *) 'Error: negative real or zero argument for', &
385 ' natural logarithm in expression ', trim(expr)
386 write (*, *)
387 return
388 end if
389 cval = log(oper1)
390 case ('LOG10')
391 if (real(oper1, r8) <= 0. .and. aimag(oper1) == 0.) then
392 if (present(ierr)) ierr = 8
393 write (*, *) 'Error: negative real or zero argument for base', &
394 '10 logarithm in expression ', trim(expr)
395 write (*, *)
396 return
397 end if
398 cval = log(oper1)/2.30258509299405_r8
399 case ('EXP')
400 cval = exp(oper1)
401 case ('COMPLEX')
402 if (itop == 0) then
403 if (present(ierr)) ierr = 7
404 write (*, *) 'Error in expression ', trim(expr)
405 write (*, *) 'Missing argument(s) for function ', trim(x%char)
406 write (*, *)
407 return
408 else
409 call popval(oper2) ! Pull second argument off stack
410 end if
411 valr = real(oper2, r8)
412 vali = real(oper1, r8)
413 cval = cmplx(valr, vali, c8)
414 case ('CONJG')
415 cval = conjg(oper1)
416 case ('ANG')
417 cval = atan2(aimag(oper1), real(oper1, r8))
418 case ('REAL')
419 cval = real(oper1, r8)
420 case ('IMAG')
421 cval = aimag(oper1)
422 case default ! Undefined function
423 if (present(ierr)) ierr = 13
424 write (*, *) 'Error: the function ', trim(x%char), ' is undefined', &
425 ' in the expression ', trim(expr)
426 write (*, *)
427 return
428 end select
429 call pushval(cval) ! Put result back on stack
430 end select
431 end do
432
433 end subroutine

515 subroutine get_next_token(str, tok, icp, isp)
516 character(*) :: str
517 character :: cop, chtemp
518 type(item) :: tok
519 integer :: icp, isp, inext, ipos, lstr, ntok
520
521 lstr = len_trim(str)
522 if (lstr == 0) then
523 tok%char = '#' ! Output end token
524 tok%type = 'E'
525 return
526 end if
527 ipos = scan(str, '+-*/^(),') ! Look for an arithmetic operator
528 ! + - * / ^ ( ) or ,
529 if (ipos > 0) cop = str(ipos:ipos)
530 select case (ipos)
531 case (0) ! Operators not present
532 ntok = ntok + 1
533 tok%char = str
534 tok%type = 'S'
535 str = ''
536 icp = 0
537 isp = 0
538 case (1)
539 tok%char = cop
540 select case (cop)
541 case ('+', '-')
542 if (ibin == 0) then
543 tok%type = 'U'
544 icp = 4
545 isp = 3
546 else
547 tok%type = 'B'
548 icp = 1
549 isp = 1
550 end if
551 ibin = 0
552 case ('*', '/')
553 tok%type = 'B'
554 icp = 2
555 isp = 2
556 ibin = 0
557 case ('^')
558 tok%type = 'B'
559 icp = 4
560 isp = 3
561 ibin = 0
562 case ('(')
563 tok%type = 'L'
564 icp = 4
565 isp = 0
566 ibin = 0
567 case (')')
568 tok%type = 'R'
569 icp = 0
570 isp = 0
571 ibin = 1
572 case (',')
573 tok%type = 'D'
574 icp = 0
575 isp = 0
576 ibin = 0
577 end select
578 str = str(2:)
579 case (2:)
580 select case (cop)
581 case ('(')
582 tok%char = str(1:ipos - 1)
583 tok%type = 'F'
584 icp = 4
585 isp = 0
586 ibin = 0
587 str = str(ipos:)
588 case ('+', '-')
589 chtemp = uppercase(str(ipos - 1:ipos - 1))
590 if (is_letter(str(1:1)) .or. chtemp /= 'E') then
591 tok%char = str(1:ipos - 1)
592 tok%type = 'S'
593 icp = 0
594 isp = 0
595 ibin = 1
596 str = str(ipos:)
597 else
598 inext = scan(str(ipos + 1:), '+-*/^(),')
599 if (inext == 0) then
600 tok%char = str
601 tok%type = 'S'
602 icp = 0
603 isp = 0
604 ibin = 0
605 str = ''
606 else
607 tok%char = str(1:ipos + inext - 1)
608 tok%type = 'S'
609 icp = 0
610 isp = 0
611 ibin = 1
612 str = str(ipos + inext:)
613 end if
614 end if
615 case default
616 tok%char = str(1:ipos - 1)
617 tok%type = 'S'
618 icp = 0
619 isp = 0
620 ibin = 1
621 str = str(ipos:)
622 end select
623 end select
624 end subroutine

628 subroutine removesp(str)
629 character(*) :: str
630 !private
631 character(1) :: ch
632 character(len_trim(str)) :: outstr
633 integer :: i, k, ich, lenstr
634
635 str = adjustl(str)
636 lenstr = len_trim(str)
637 outstr = ' '
638 k = 0
639
640 do i = 1, lenstr
641 ch = str(i:i)
642 ich = iachar(ch)
643 select case (ich)
644 case (0:32) ! space, tab, or control character
645 cycle
646 case (33:)
647 k = k + 1
648 outstr(k:k) = ch
649 end select
650 end do
651 str = replacestr(outstr, '**', '^')
652 contains
653 pure recursive function replacestr(string, search, substitute) result(outstr)
654 character(*), intent(in) :: string, search, substitute
655 character(:), allocatable :: outstr
656 !private
657 integer :: i, lenstr, lensearch
658
659 lenstr = len(string)
660 lensearch = len(search)
661 if (lenstr == 0 .or. lensearch == 0) then
662 outstr = ''
663 return
664 elseif (lenstr < lensearch) then
665 outstr = string
666 return
667 end if
668 i = 1
669 do
670 if (string(i:i + lensearch - 1) == search) then
671 outstr = string(1:i - 1)//substitute//&
672 replacestr(string(i + lensearch:lenstr), search, substitute)
673 exit
674 end if
675 if (i + lensearch > lenstr) then
676 outstr = string
677 exit
678 end if
679 i = i + 1
680 cycle
681 end do
682 end function
683 end subroutine
