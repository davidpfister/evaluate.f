63 complex(c8), intent(in) :: val
64 integer, intent(out), optional :: ierr
65 !private
66 character(len_trim(sym)) :: usym
67 integer :: i
68 if (present(ierr)) ierr = 0
69
70 call setdefault()
71
72 ! Assign val to sym if sym is already in symbol table
73 usym = uppercase(sym)
74 if (.not. is_letter(sym(1:1)) .or. len_trim(sym) > 24) then
75 if (present(ierr)) ierr = 11
76 write (*, *) 'Error: symbol ', trim(sym), ' has improper format'
77 write (*, *)
78 return
79 end if

164 subroutine getparam(sym, var, ierr)
165 character(*), intent(inout) :: sym
166 complex(c8), intent(out) :: var
167 integer, intent(out), optional :: ierr
168 !private
169 character(len_trim(sym)) :: usym
170 integer :: ifind, j
171
172 if (present(ierr)) ierr = 0
173 sym = adjustl(sym)
174 if (.not. is_letter(sym(1:1)) .or. len_trim(sym) > 24) then
175 if (present(ierr)) ierr = 11
176 write (*, *) 'Error: symbol ', trim(sym), ' has incorrect format'
177 write (*, *)
178 return
179 end if
180 ifind = 0
181 usym = uppercase(sym)
182 do j = 1, nparams
183 if (trim(usym) == trim(params(j)%symbol)) then
184 var = params(j)%val
185 ifind = j
186 exit
187 end if
188 end do
189 if (ifind == 0) then
190 if (present(ierr)) ierr = 4
191 write (*, *) 'Error: symbol ', trim(sym), ' not in symbol table'
192 write (*, *)
193 return
194 end if
195 end subroutine

207 subroutine getvalue(xinchar, cval, ierr)
208 character(*), intent(inout) :: xinchar
209 complex(c8), intent(out) :: cval
210 integer, intent(out), optional :: ierr
211 !private
212 real(r8) :: rval
213 integer :: ios
214
215 if (present(ierr)) ierr = 0
216
217 if (is_letter(xinchar(1:1))) then ! xinchar is a symbol
218 call getparam(xinchar, cval)
219 else ! xinchar is a number string
220 call value(xinchar, rval, ios) ! rval is the value of xinchar
221 if (ios > 0) then
222 if (present(ierr)) ierr = 3
223 write (*, *) 'Error: number string ', trim(xinchar), ' does not correspond to a valid number'
224 write (*, *)
225 end if
226 cval = cmplx(rval, 0.0_r8, c8)
227 return
228 end if
229 end subroutine
